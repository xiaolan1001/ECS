255 size_t node_word_get_width(node_ptr w);
256 
257 #define _CHECK_WORD(w) \
258  nusmv_assert(((node_get_type(w) == UNSIGNED_WORD || \
259  node_get_type(w) == SIGNED_WORD) && \
260  node_word_get_width(w) > 0) || \
261  (node_get_type(w) == NUMBER_UNSIGNED_WORD || \
262  node_get_type(w) == NUMBER_SIGNED_WORD))
263 
264 
265 #define _CHECK_WORDS(w1, w2) \
266  _CHECK_WORD(w1); _CHECK_WORD(w2); \
267  if ((node_get_type(w1) == UNSIGNED_WORD || \
268  node_get_type(w1) == SIGNED_WORD) && \
269  (node_get_type(w2) == UNSIGNED_WORD || \
270  node_get_type(w2) == SIGNED_WORD)) /* all words */ \
271  nusmv_assert(node_word_get_width(w1) == node_word_get_width(w2)); \
272  else if ((node_get_type(w1) == UNSIGNED_WORD || \
273  node_get_type(w1) == SIGNED_WORD) && \
274  (node_get_type(w2) == NUMBER_UNSIGNED_WORD || \
275  node_get_type(w2) == NUMBER_SIGNED_WORD)) /* word and const */ \
276  nusmv_assert(node_word_get_width(w1) == \
277  WordNumber_get_width(WORD_NUMBER(car(w2)))); \
278  else if ((node_get_type(w2) == UNSIGNED_WORD || \
279  node_get_type(w2) == SIGNED_WORD) && \
280  (node_get_type(w1) == NUMBER_UNSIGNED_WORD || \
281  node_get_type(w1) == NUMBER_SIGNED_WORD)) /* const and word */ \
282  nusmv_assert(node_word_get_width(w2) == \
283  WordNumber_get_width(WORD_NUMBER(car(w1)))); \
284  else if ((node_get_type(w2) == NUMBER_UNSIGNED_WORD || \
285  node_get_type(w2) == NUMBER_SIGNED_WORD) && \
286  (node_get_type(w1) == NUMBER_UNSIGNED_WORD || \
287  node_get_type(w1) == NUMBER_SIGNED_WORD)) /* const and const */ \
288  nusmv_assert(WordNumber_get_width(WORD_NUMBER(car(w2))) == \
289  WordNumber_get_width(WORD_NUMBER(car(w1)))); \
290  else error_unreachable_code();
