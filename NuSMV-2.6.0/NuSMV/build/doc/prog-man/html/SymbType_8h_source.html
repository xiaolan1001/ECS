148 typedef enum SymbTypeTag_TAG {

150  SYMB_TYPE_STATEMENT, /* for statements like assignements, INIT, TRANS etc */
151  SYMB_TYPE_BOOLEAN,
152  SYMB_TYPE_ENUM, /* an enumeration of values (includes ranges, i.e. -1..4) */
153  SYMB_TYPE_INTEGER, /* (infinite-precision) integer */
154  SYMB_TYPE_REAL, /* (infinite-precision) rational */
155  SYMB_TYPE_NFUNCTION,
156  SYMB_TYPE_CONTINUOUS,
157  SYMB_TYPE_SIGNED_WORD, /* word is like an arrary of booleans + signed arithmetic */
158  SYMB_TYPE_UNSIGNED_WORD, /* word is like an arrary of booleans + unsigned arithmetic */
159  SYMB_TYPE_WORDARRAY, /* an array of words */
160  SYMB_TYPE_ARRAY, /* an array */
161  SYMB_TYPE_SET_BOOL, /* a set of boolean values */
162  SYMB_TYPE_SET_INT, /* a set of integer values */
163  SYMB_TYPE_SET_SYMB, /* a set of symbolic values */
164  SYMB_TYPE_SET_INT_SYMB, /* a set of symbolic and integer values */
165  SYMB_TYPE_INTARRAY, /* unbounded array with integer index */
166  SYMB_TYPE_ERROR, /* indicates an error */
167  /* SYMB_TYPE_NONE must be the first and SYMB_TYPE_ERROR must be

170 } SymbTypeTag;
