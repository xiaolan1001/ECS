99 /* Constants for setting and clearing pointer annotation bit which is
100  lowest (rightmost) bit.
101 */
102 #if !defined(NUSMV_SIZEOF_VOID_P) || !defined(NUSMV_SIZEOF_LONG)
103 #error Constants NUSMV_SIZEOF_VOID_P and NUSMV_SIZEOF_LONG must be defined
104 #endif
105 
106 #if (NUSMV_SIZEOF_VOID_P == NUSMV_SIZEOF_LONG)
107 /* The lowest bit of a pointer is used to store additional info
108  (about complementation). We can use this bit only because pointers
109  never use it (it is always 0 for any pointer). This is
110  ensured by the alignment of pointers which is required to be at
111  least 2 bytes.
112 
113  Note: in past the highest (leftmost) bit was used as the annotation
114  bit. But it appeared that if the library is invoked from Java on
115  multi-processor machine then malloc behaves differently and it is
116  possible for a pointer to be from the higher half of memory, i.e.
117  in memory address the highest bit could be 0 as well as 1. As
118  result we could not use it any more and changed the
119  implementation.
120 
121  Current implementation is similar to CUDD's one.
122 
123  Warning: on most machine alignment is 4 bytes which is more than
124  enough for us. However in some case additional compilation options
125  (e.g. -malign-double) may be required to force proper alignment.
126 */
127 # define DAG_ANNOTATION_BIT ((nusmv_ptrint) 1)
128 #else
129 /* this fallback setting is for 32-bit only */
130 /* Size of pointers and long are different (may hide a serious problem) */
131 # define DAG_ANNOTATION_BIT ((nusmv_ptrint) 1)
132 #endif
