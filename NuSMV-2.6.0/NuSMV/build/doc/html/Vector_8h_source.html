173  { \

197  { \

272  static void prefix ## _sort_impl( \
273  type *arr, size_t size, \
274  int (*cmp)(type a, type b, void *p), void *p) \
275  { \
276  /* sorting algorithm taken from minisat */ \
277  if (size <= 15) { \
278  /* selection sort */ \
279  size_t i, j, best_i; \
280  type tmp; \
281  \
282  for (i = 1; i < size; ++i){ \
283  best_i = i-1; \
284  for (j = i; j < size; ++j) { \
285  if (cmp(arr[j], arr[best_i], p) < 0) { \
286  best_i = j; \
287  } \
288  } \
289  tmp = arr[i-1]; \
290  arr[i-1] = arr[best_i]; \
291  arr[best_i] = tmp; \
292  } \
293  } else { \
294  size_t i, j; \
295  type pivot; \
296  type tmp; \
297  pivot = arr[size / 2]; \
298  i = (size_t)-1; \
299  j = size; \
300  \
301  for (;;) { \
302  do i++; while(cmp(arr[i], pivot, p) < 0); \
303  do j--; while(cmp(pivot, arr[j], p) < 0); \
304  \
305  if (i >= j) break; \
306  \
307  tmp = arr[i]; \
308  arr[i] = arr[j]; \
309  arr[j] = tmp; \
310  } \
311  \
312  prefix ## _sort_impl(arr, i, cmp, p); \
313  prefix ## _sort_impl(&arr[i], size-i, cmp, p); \
314  } \
315  } \

321  } \
